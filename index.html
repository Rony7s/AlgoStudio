Bubble Sort

                    
#include<iostream>
using namespace std;

int main(){
    int n;
    cout <<"Enter Array Size: "<<endl;
    cin >>n;
    int arr[n];

    for(int i=0; i<n; i++){
        cout <<"Enter Array element of index: "<< i <<endl;
        cin>>arr[i];
    }
    //bubble Sort
    for(int i=0; i<n; i++){
        for(int j=0; j<n-i-1; j++){
            if(arr[j]>arr[j+1]){
                //swap two numbers without using third variable
                arr[j] = arr[j] + arr[j+1];
                arr[j+1] = arr[j] - arr[j+1];
                arr[j] = arr[j] - arr[j+1];
            }
        }
    }

    cout <<"The Sort result is : "<<endl;
    for(int i=0; i<n; i++){
        cout << arr[i]<<" ";
    }

    return 0;
}
